library(sf)
library(USAboundaries)
library(tidyverse)
#Step 1.1
counties = USAboundaries::us_counties() %>%
  filter(!state_name %in% c("Hawaii",
                            "Alaska",
                            "Puerto Rico",
                            "Guam")) %>%
  st_as_sf(coords = c("lng", "lat"), crs = 4326) %>%
  st_transform(5070)
#Step 1.2
countycent = st_centroid(counties) 
countycent_c = st_combine(countycent)
# Step 1.3
st_voronoi(countycent_c) 
## Geometry set for 1 feature 
## geometry type:  GEOMETRYCOLLECTION
## dimension:      XY
## bbox:           xmin: -6808247 ymin: -4159626 xmax: 6705205 ymax: 7632254
## projected CRS:  NAD83 / Conus Albers
v_grid = st_voronoi(countycent_c) %>%
  st_cast() %>% 
  st_as_sf() %>% 
  mutate(id = 1:n())

st_triangulate(countycent_c)
## Geometry set for 1 feature 
## geometry type:  GEOMETRYCOLLECTION
## dimension:      XY
## bbox:           xmin: -2303763 ymin: 344857.6 xmax: 2200721 ymax: 3127770
## projected CRS:  NAD83 / Conus Albers
tri_grid = st_triangulate(countycent_c) %>%
  st_cast() %>% 
  st_as_sf() %>% 
  mutate(id = 1:n())

sqgrid = st_make_grid(counties, n=70) %>%
  st_cast() %>% 
  st_as_sf() %>% 
  mutate(id = 1:n())
  
hexgrid = st_make_grid(counties, n=70, square=FALSE) %>%
  st_cast() %>%
  st_as_sf() %>%
  mutate(id = 1:n())
# Step 1.4

plot(v_grid)

plot(tri_grid)

plot(sqgrid)

plot(hexgrid)

countybound = st_union(counties) 

plot(countybound)
